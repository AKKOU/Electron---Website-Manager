#include<stdio.h>
#include<iostream>
#include<vector>
using namespace std;

int main() {
  long long int data[5];

  while (scanf("%lld%lld%lld%lld%lld", & data[0], & data[1], & data[2], & data[3], & data[4]) != EOF) {
    if (abs(data[0] - data[2]) + abs(data[1] - data[3]) <= data[4]) {
      printf("die\n");
    } else {
      printf("alive\n");
    }
  }
  return 0;
}
